CREATE TABLE Client(
	Numar_Telefon  varchar(15) NOT NULL PRIMARY KEY,
	Adresa varchar(100) NOT NULL,
	Nume varchar(20) NOT NULL
);

CREATE TABLE Comenzi(
	Numar_Telefon varchar(15) ,
	comanda varchar(500) NOT NULL,
	PRIMARY KEY (Numar_Telefon,comanda),
	FOREIGN KEY (Numar_Telefon) REFERENCES Client(Numar_Telefon)
);

CREATE TABLE Firma(
	Nume_Firma varchar(20) NOT NULL UNIQUE PRIMARY KEY,
	Lider varchar(20)
);

CREATE TABLE Parteneri(
	Partener_id int unique PRIMARY KEY,
	Locatie varchar(100),
	Nume_Firma varchar(20),
	Nume varchar(10) 
);

CREATE TABLE Aplicatie(
	Nume_Aplicatie varchar(20) Primary key NOT NULL,
	Nume_Firma varchar(20),
	Numar_comenzi int,
	Numar_utilizatori int,
	FOREIGN KEY (Nume_Firma) REFERENCES Firma(Nume_Firma)
);

CREATE TABLE CosCumparaturi(
	Cos_ID int UNIQUE NOT NULL PRIMARY KEY,
	Nume_Parteneri varchar(20) NOT NULL,
	Nume_Produse varchar(100) NOT NULL,
	Nume_Aplicatie varchar(20),
	FOREIGN KEY (Nume_Aplicatie) REFERENCES Aplicatie(Nume_Aplicatie)
);

CREATE TABLE Produse(
	Locatie varchar(100) NOT NULL PRIMARY KEY,
	Produs varchar(20) NOT NULL UNIQUE
);

CREATE TABLE Livratori(
	Numar_Telefon varchar(14) PRIMARY KEY UNIQUE NOT NULL,
	Nume varchar(20)
);

CREATE TABLE Vehicul(
	Numar_Inmatriculare varchar(10) NOT NULL UNIQUE PRIMARY KEY,
	Producator varchar(10) NOT NULL,
	Model  varchar (10),
	Model_Full_Name varchar(20)
);

CREATE TABLE TaraProducator(
	Producator varchar(10) NOT NULL PRIMARY KEY,
	Tara_Producator varchar(15) NOT NULL
);
